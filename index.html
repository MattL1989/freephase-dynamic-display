function normaliseDay(dayObj) { const d = dayObj || {}; const amber = d.amber || null; const red = d.red || null; let morningGreen = d.green_morning || null; let lateGreen = d.green_late || null; const g = d.green || null; if (g) { const startH = new Date(g.start).getUTCHours(); const endH = new Date(g.end).getUTCHours(); if (!morningGreen && endH === 6) { morningGreen = g; } if (!lateGreen && startH === 23) { lateGreen = g; } } return { morningGreen, lateGreen, amber, red }; }
